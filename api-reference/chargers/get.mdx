---
title: 'Get Chargers'
api: 'GET /teams/{team_id}/chargers'
description: 'Returns all the chargers that the specified team has access to'
---
## Request
<Tabs>
  <Tab title="Path">
    <ParamField path="team_id" type="integer" required>
      The id of the team
    </ParamField>
  </Tab>
  <Tab title="Query">
    <ParamField query="page" type="integer">
      The page number to return
    </ParamField>
  </Tab>
</Tabs>

## Response
<Tabs>
  <Tab title="Data">
    <ResponseField name="data" type="Charger[]">
      Array of chargers
      <Expandable title="Charger">
        <ResponseField name="id" type="integer">
          The identifier of the charger
        </ResponseField>
        <ResponseField name="identity" type="string">
          The identity of the charger
        </ResponseField>
        <ResponseField name="connection_status" type="ConnectionStatus">
          The connection status of the charger
          <Expandable title="ConnectionStatus">
            `never_seen`
            `offline`
            `online`
          </Expandable>
        </ResponseField>
        <ResponseField name="status" type="?Status">
          The status of the charger
          <Expandable title="Status">
            `Available`
            `Unavailable`
            `Faulted`
          </Expandable>
        </ResponseField>
        <ResponseField name="error" type="?Error">
          The error status of the charger
          <Expandable title="Error">
            `ConnectorLockFailure`
            `EVCommunicationError`
            `GroundFailure`
            `HighTemperature`
            `InternalError`
            `LocalListConflict`
            `NoError`
            `OtherError`
            `OverCurrentFailure`
            `OverVoltage`
            `PowerMeterFailure`
            `PowerSwitchFailure`
            `ReaderFailure`
            `ResetFailure`
            `UnderVoltage`
            `WeakSignalLevel`
          </Expandable>
        </ResponseField>
        <ResponseField name="error_info" type="?string">
          Extra error information for the charger
        </ResponseField>
        <ResponseField name="created_at" type="string">
          The date the charger was created
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          The date the charger was last updated
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Tab>
