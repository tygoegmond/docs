---
title: 'Create a new team'
api: 'POST /teams'
description: 'Creates a new team for the user'
---

## Request
### Body
<ParamField body="name" type="string" required>
  The name of the new team
</ParamField>
<ParamField body="email" type="string" required>
  The email of the new team
</ParamField>
<ParamField body="type" type="TeamType" required>
  The type of the new team
</ParamField>

## Response
<ResponseField name="data" type="Team">
  The newly created team
  <Expandable title="Team">
    <ResponseField name="id" type="integer">
      The identifier of the team
    </ResponseField>
    <ResponseField name="name" type="string">
      The name of the team
    </ResponseField>
    <ResponseField name="type" type="TeamType">
      The type of the team
      <Expandable title="TeamType">
        `Personal`
        `HOA`
        `Retailer`
        `Installer`
        `Wholesaler`
        `Company`
        `OEM`
        `Other`
      </Expandable>
    </ResponseField>
    <ResponseField name="email" type="string">
      The email of the team
    </ResponseField>
    <ResponseField name="created_at" type="string">
      The date the team was created
    </ResponseField>
    <ResponseField name="updated_at" type="string">
      The date the team was last updated
    </ResponseField>
  </Expandable>
</ResponseField>
