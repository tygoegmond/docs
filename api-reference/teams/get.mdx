---
title: 'Get Teams'
api: 'GET /teams'
description: 'Returns all teams from the system that the user has access to'
---

## Response
<Tabs>
  <Tab title="Data">
    <ResponseField name="data" type="Team[]">
      Array of teams
      <Expandable title="Team">
        <ResponseField name="id" type="integer">
          The identifier of the team
        </ResponseField>
        <ResponseField name="name" type="string">
          The name of the team
        </ResponseField>
        <ResponseField name="type" type="TeamType">
          The type of the team
          <Expandable title="TeamType">
            `Personal`
            `HOA`
            `Retailer`
            `Installer`
            `Wholesaler`
            `Company`
            `OEM`
            `Other`
          </Expandable>
        </ResponseField>
        <ResponseField name="email" type="string">
          The email of the team
        </ResponseField>
        <ResponseField name="created_at" type="string">
          The date the team was created
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          The date the team was last updated
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Tab>
